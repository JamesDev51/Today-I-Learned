abstract class
```typescript
abstract class User{
        constructor(
        private firstName:string,
        private lastNmae:string,
        private nickname:string
    ){}
    abstract getNickname():void //추상메서드(구현 x)
    getFullName(){
        return `${this.firstName} ${this.lastName}`
    }
}

class Player extends User {
    getNickname(){
        console.log(this.nickname) //상속해도 private 접근은 불가능
    }
}


const james = new User("minseok", "jeong", "james") //불가능 (추상클래스)
const james = new Player("minseok", "jeong", "james")
```

```typescript
type Words = {
    [key:string]: string
}

class Dict {
    private words: Words
    constructor(){
        this.words = {}
    }
    add(word:Word){
        if(this.words[word.term] === undefined){
            this.words[word.term] = word.def
        }
    }
    def(term:string){
        return this.words[term]
    }
}

class Words {
    constructor(
        public term:string,
        public def string
    )
}

const kimchi = new Word("kimchi", "한국 음식")

const dict = new Dict()

dict.add(kimchi)
dict.def("kimchi")
```
